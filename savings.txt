    var xhr;
    var category, $category;
    var subcategory, $subcategory;
    var type, $type;

    $category = $('#category').selectize({
        onChange: function(value) {
            if (!value.length) return;
            type.disable();
            $('#type_wrap').addClass('hidden');
            subcategory.disable();
            subcategory.clearOptions();
            subcategory.load(function(callback) {
                xhr && xhr.abort();
                xhr = $.ajax({
                    url: '/Bootstrap/json/cat_' + value + '.json',
                    type: 'GET',
                    dataType: 'json',
                    success: function(results) {
                        subcategory.enable();
                        callback(results);
                    },
                    error: function() {
                        callback();
                    }
                })
            });
        }
    });

    $subcategory = $('#subcategory').selectize({
        valueField: 'name',
        labelField: 'name',
        searchField: ['name'],
        onChange: function(value) {
            if (!value.length) return;
            type.disable();
            type.clearOptions();
            type.load(function(callback) {
                xhr && xhr.abort();
                xhr = $.ajax({
                    url: '/Bootstrap/json/type.json',
                    type: 'GET',
                    dataType: 'json',
                    success: function(results) {
                        type.enable();
                        $('#type_wrap').removeClass('hidden');
                        callback(results);
                    },
                    error: function() {
                        callback();
                    }
                })
            });
        }
    });
    $type = $('#type').selectize({
        valueField: 'name',
        labelField: 'name',
        searchField: ['name']
    });

    subcategory  = $subcategory[0].selectize;
    category = $category[0].selectize;
    type = $type[0].selectize;

    subcategory.disable();
});
